#### 2.1 创建建表脚本 `src/schema/create_tables.sql`

```sql
-- ============================================
-- 图书管理系统 - 建表脚本
-- ============================================

-- 开启外键约束
PRAGMA foreign_keys = ON;

-- 创建作者表
CREATE TABLE IF NOT EXISTS authors (
    author_id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    birth_year INTEGER,
    nationality TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 创建书籍表
CREATE TABLE IF NOT EXISTS books (
    book_id INTEGER PRIMARY KEY AUTOINCREMENT,
    title TEXT NOT NULL,
    author_id INTEGER NOT NULL,
    isbn TEXT UNIQUE,
    category TEXT,
    price REAL CHECK(price >= 0),
    stock INTEGER DEFAULT 1,
    published_year INTEGER,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (author_id) REFERENCES authors(author_id) ON DELETE CASCADE
);

-- 创建读者表
CREATE TABLE IF NOT EXISTS readers (
    reader_id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    email TEXT UNIQUE NOT NULL,
    phone TEXT,
    register_date DATE DEFAULT CURRENT_DATE,
    status TEXT DEFAULT 'active' CHECK(status IN ('active', 'suspended', 'expired'))
);

-- 创建借阅记录表
CREATE TABLE IF NOT EXISTS borrow_records (
    record_id INTEGER PRIMARY KEY AUTOINCREMENT,
    book_id INTEGER NOT NULL,
    reader_id INTEGER NOT NULL,
    borrow_date DATE DEFAULT CURRENT_DATE,
    due_date DATE,
    return_date DATE,
    status TEXT DEFAULT 'borrowed' CHECK(status IN ('borrowed', 'returned', 'overdue')),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (book_id) REFERENCES books(book_id),
    FOREIGN KEY (reader_id) REFERENCES readers(reader_id)
);

-- 创建触发器：自动设置应还日期（借书后14天）
CREATE TRIGGER IF NOT EXISTS set_due_date 
AFTER INSERT ON borrow_records
WHEN NEW.due_date IS NULL
BEGIN
    UPDATE borrow_records 
    SET due_date = DATE('now', '+14 days') 
    WHERE record_id = NEW.record_id;
END;

-- 创建触发器：自动更新逾期状态
CREATE TRIGGER IF NOT EXISTS update_overdue_status
AFTER UPDATE ON borrow_records
WHEN NEW.return_date IS NULL AND NEW.due_date < DATE('now')
BEGIN
    UPDATE borrow_records 
    SET status = 'overdue' 
    WHERE record_id = NEW.record_id;
END;

-- 创建索引
CREATE INDEX IF NOT EXISTS idx_books_title ON books(title);
CREATE INDEX IF NOT EXISTS idx_books_author ON books(author_id);
CREATE INDEX IF NOT EXISTS idx_borrow_records_reader ON borrow_records(reader_id);
CREATE INDEX IF NOT EXISTS idx_borrow_records_status ON borrow_records(status);
CREATE INDEX IF NOT EXISTS idx_borrow_records_dates ON borrow_records(borrow_date, due_date);
```

**`src/queries/join_queries.sql`**
```sql
-- ============================================
-- 连接查询练习
-- ============================================

-- 1. 内连接 - 查询书籍及其作者
SELECT b.title, a.name AS author, b.category, b.price
FROM books b
INNER JOIN authors a ON b.author_id = a.author_id;

-- 2. 左连接 - 查询所有书籍及借阅状态
SELECT b.title, br.status, r.name AS borrower
FROM books b
LEFT JOIN borrow_records br ON b.book_id = br.book_id
LEFT JOIN readers r ON br.reader_id = r.reader_id;

-- 3. 三表连接 - 完整的借阅信息
SELECT 
    r.name AS 读者,
    b.title AS 书名,
    a.name AS 作者,
    br.borrow_date AS 借书日期,
    br.due_date AS 应还日期,
    br.status AS 状态
FROM borrow_records br
JOIN readers r ON br.reader_id = r.reader_id
JOIN books b ON br.book_id = b.book_id
JOIN authors a ON b.author_id = a.author_id
ORDER BY br.borrow_date DESC;

-- 4. 自连接 - 查询同一作者的其他书籍
SELECT b1.title AS 书籍1, b2.title AS 书籍2
FROM books b1
JOIN books b2 ON b1.author_id = b2.author_id AND b1.book_id < b2.book_id;

-- 5. 查询未归还的书籍
SELECT b.title, r.name AS 借阅者, br.due_date
FROM borrow_records br
JOIN books b ON br.book_id = b.book_id
JOIN readers r ON br.reader_id = r.reader_id
WHERE br.status IN ('borrowed', 'overdue');
```

**`src/queries/aggregate.sql`**
```sql
-- ============================================
-- 聚合查询练习
-- ============================================

-- 1. 统计书籍总数
SELECT COUNT(*) AS total_books FROM books;

-- 2. 按分类统计
SELECT category, COUNT(*) AS count, AVG(price) AS avg_price
FROM books
GROUP BY category;

-- 3. 每个作者的书籍数量
SELECT a.name, COUNT(b.book_id) AS book_count
FROM authors a
LEFT JOIN books b ON a.author_id = b.author_id
GROUP BY a.author_id
ORDER BY book_count DESC;

-- 4. 读者借阅数量统计
SELECT r.name, COUNT(br.record_id) AS borrow_count
FROM readers r
LEFT JOIN borrow_records br ON r.reader_id = br.reader_id
GROUP BY r.reader_id
ORDER BY borrow_count DESC;

-- 5. 当前未归还书籍统计
SELECT status, COUNT(*) AS count
FROM borrow_records
GROUP BY status;

-- 6. HAVING - 借阅超过2本的读者
SELECT r.name, COUNT(br.record_id) AS borrow_count
FROM readers r
JOIN borrow_records br ON r.reader_id = br.reader_id
GROUP BY r.reader_id
HAVING COUNT(br.record_id) >= 2;

-- 7. 统计各分类的库存总值
SELECT category, SUM(price * stock) AS total_value
FROM books
GROUP BY category;

-- 8. 统计每月借阅数量
SELECT strftime('%Y-%m', borrow_date) AS month, COUNT(*) AS borrow_count
FROM borrow_records
GROUP BY strftime('%Y-%m', borrow_date)
ORDER BY month;
```

#### 2.4 创建事务脚本

**`src/transactions/borrow_transaction.sql`**
```sql
-- ============================================
-- 借书事务
-- ============================================

-- 开始事务
BEGIN TRANSACTION;

-- 变量定义（在SQLite中可以用这种方式）
-- 假设借阅: 读者ID=3, 书籍ID=1

-- 1. 检查书籍库存
SELECT stock FROM books WHERE book_id = 1;

-- 2. 如果库存>0，插入借阅记录
INSERT INTO borrow_records (book_id, reader_id) 
VALUES (1, 3);

-- 3. 更新库存
UPDATE books 
SET stock = stock - 1 
WHERE book_id = 1 AND stock > 0;

-- 4. 检查影响的行数
SELECT changes() AS rows_affected;

-- 5. 如果成功，提交事务
COMMIT;

-- 如果失败，回滚
-- ROLLBACK;
```

**`src/transactions/return_transaction.sql`**
```sql
-- ============================================
-- 还书事务
-- ============================================

BEGIN TRANSACTION;

-- 假设归还: 记录ID=3
-- 1. 更新借阅记录
UPDATE borrow_records 
SET return_date = CURRENT_DATE, 
    status = 'returned' 
WHERE record_id = 3;

-- 2. 更新书籍库存
UPDATE books 
SET stock = stock + 1 
WHERE book_id = (SELECT book_id FROM borrow_records WHERE record_id = 3);

-- 3. 检查是否成功
SELECT changes();

COMMIT;
```

#### 3.1 Linux/Mac 初始化脚本 `scripts/init_db.sh`

```bash
#!/bin/bash

# ============================================
# 图书管理系统 - 数据库初始化脚本
# ============================================

echo "📚 开始初始化图书管理系统数据库..."

# 设置数据库文件路径
DB_PATH="../data/library.db"

# 切换到脚本目录
cd "$(dirname "$0")"

# 删除旧的数据库文件（如果存在）
if [ -f "$DB_PATH" ]; then
    echo "🗑️  删除旧的数据库文件..."
    rm "$DB_PATH"
fi

# 创建数据库并执行所有脚本
echo "🔨 创建数据库表..."
sqlite3 "$DB_PATH" < ../src/schema/create_tables.sql

echo "📝 插入作者数据..."
sqlite3 "$DB_PATH" < ../src/data/insert_authors.sql

echo "📝 插入书籍数据..."
sqlite3 "$DB_PATH" < ../src/data/insert_books.sql

echo "📝 插入读者数据..."
sqlite3 "$DB_PATH" < ../src/data/insert_readers.sql

echo "📝 插入借阅记录..."
sqlite3 "$DB_PATH" < ../src/data/insert_borrows.sql

echo "✅ 数据库初始化完成！"
echo "📊 数据库位置: $DB_PATH"

# 显示数据库信息
echo ""
echo "📈 数据库统计信息："
sqlite3 "$DB_PATH" <<EOF
SELECT '📖 书籍总数: ' || COUNT(*) FROM books;
SELECT '👤 作者总数: ' || COUNT(*) FROM authors;
SELECT '👥 读者总数: ' || COUNT(*) FROM readers;
SELECT '📋 借阅记录: ' || COUNT(*) FROM borrow_records;
EOF

echo ""
echo "💡 使用方法："
echo "   sqlite3 $DB_PATH"
echo "   或运行: sqlite3 $DB_PATH < src/queries/basic_queries.sql"
```

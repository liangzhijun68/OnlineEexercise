## 📚 图书管理系统 - 完整案例

### 1️⃣ 数据库设计

```sql
-- ============================================
-- 图书管理系统 SQLite3 数据库搭建
-- 包含：书籍、作者、读者、借阅记录 四张表
-- ============================================

-- 开启外键约束 (SQLite默认关闭)
PRAGMA foreign_keys = ON;

-- ============================================
-- 1. 创建表 (CREATE TABLE)
-- ============================================

-- 作者表
CREATE TABLE authors (
    author_id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    birth_year INTEGER,
    nationality TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 书籍表
CREATE TABLE books (
    book_id INTEGER PRIMARY KEY AUTOINCREMENT,
    title TEXT NOT NULL,
    author_id INTEGER NOT NULL,
    isbn TEXT UNIQUE,
    category TEXT,
    price REAL CHECK(price >= 0),
    stock INTEGER DEFAULT 1,
    published_year INTEGER,
    FOREIGN KEY (author_id) REFERENCES authors(author_id) ON DELETE CASCADE
);

-- 读者表
CREATE TABLE readers (
    reader_id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    email TEXT UNIQUE NOT NULL,
    phone TEXT,
    register_date DATE DEFAULT CURRENT_DATE,
    status TEXT DEFAULT 'active' CHECK(status IN ('active', 'suspended', 'expired'))
);

-- 借阅记录表
CREATE TABLE borrow_records (
    record_id INTEGER PRIMARY KEY AUTOINCREMENT,
    book_id INTEGER NOT NULL,
    reader_id INTEGER NOT NULL,
    borrow_date DATE DEFAULT CURRENT_DATE,
    due_date DATE,
    return_date DATE,
    status TEXT DEFAULT 'borrowed' CHECK(status IN ('borrowed', 'returned', 'overdue')),
    FOREIGN KEY (book_id) REFERENCES books(book_id),
    FOREIGN KEY (reader_id) REFERENCES readers(reader_id)
);

-- 创建索引 (CREATE INDEX)
CREATE INDEX idx_books_title ON books(title);
CREATE INDEX idx_borrow_records_reader ON borrow_records(reader_id);
CREATE INDEX idx_borrow_records_status ON borrow_records(status);
```

### 3️⃣ 基础查询练习 (SELECT)

```sql
-- ============================================
-- 练习1: 基础查询
-- ============================================

-- 1.1 查询所有书籍
SELECT * FROM books;

-- 1.2 查询特定列
SELECT title, price, stock FROM books;

-- 1.3 条件查询 - 价格小于30的书籍
SELECT title, price, category 
FROM books 
WHERE price < 30;

-- 1.4 模糊查询 - 标题包含"哈利"的书籍
SELECT * FROM books 
WHERE title LIKE '%哈利%';

-- 1.5 排序 - 按价格降序
SELECT title, price 
FROM books 
ORDER BY price DESC;

-- 1.6 限制结果 - 前3本最贵的书
SELECT title, price 
FROM books 
ORDER BY price DESC 
LIMIT 3;

-- 1.7 去重 - 查询有哪些分类
SELECT DISTINCT category FROM books;
```

### 4️⃣ 多表连接查询 (JOIN)

```sql
-- ============================================
-- 练习2: 多表连接
-- ============================================

-- 2.1 内连接 - 查询书籍及其作者
SELECT b.title, a.name AS author, b.category, b.price
FROM books b
INNER JOIN authors a ON b.author_id = a.author_id;

-- 2.2 左连接 - 查询所有书籍及借阅状态
SELECT b.title, br.status, r.name AS borrower
FROM books b
LEFT JOIN borrow_records br ON b.book_id = br.book_id
LEFT JOIN readers r ON br.reader_id = r.reader_id;

-- 2.3 三表连接 - 完整的借阅信息
SELECT 
    r.name AS 读者,
    b.title AS 书名,
    a.name AS 作者,
    br.borrow_date AS 借书日期,
    br.due_date AS 应还日期,
    br.status AS 状态
FROM borrow_records br
JOIN readers r ON br.reader_id = r.reader_id
JOIN books b ON br.book_id = b.book_id
JOIN authors a ON b.author_id = a.author_id
ORDER BY br.borrow_date DESC;

-- 2.4 自连接 - 查询同一作者的其他书籍(示例)
SELECT b1.title AS 书籍1, b2.title AS 书籍2
FROM books b1
JOIN books b2 ON b1.author_id = b2.author_id AND b1.book_id < b2.book_id;
```

### 5️⃣ 聚合查询 (GROUP BY & 聚合函数)

```sql
-- ============================================
-- 练习3: 聚合查询
-- ============================================

-- 3.1 统计书籍总数
SELECT COUNT(*) AS total_books FROM books;

-- 3.2 按分类统计书籍数量
SELECT category, COUNT(*) AS count, AVG(price) AS avg_price
FROM books
GROUP BY category;

-- 3.3 每个作者的书籍数量
SELECT a.name, COUNT(b.book_id) AS book_count
FROM authors a
LEFT JOIN books b ON a.author_id = b.author_id
GROUP BY a.author_id
ORDER BY book_count DESC;

-- 3.4 读者借阅数量统计
SELECT r.name, COUNT(br.record_id) AS borrow_count
FROM readers r
LEFT JOIN borrow_records br ON r.reader_id = br.reader_id
GROUP BY r.reader_id
ORDER BY borrow_count DESC;

-- 3.5 当前未归还书籍统计
SELECT status, COUNT(*) AS count
FROM borrow_records
GROUP BY status;

-- 3.6 HAVING - 借阅超过2本的读者
SELECT r.name, COUNT(br.record_id) AS borrow_count
FROM readers r
JOIN borrow_records br ON r.reader_id = br.reader_id
GROUP BY r.reader_id
HAVING COUNT(br.record_id) >= 2;
```

### 6️⃣ 子查询与高级查询

```sql
-- ============================================
-- 练习4: 子查询和高级查询
-- ============================================

-- 4.1 查询最贵的书
SELECT title, price 
FROM books 
WHERE price = (SELECT MAX(price) FROM books);

-- 4.2 查询比平均价格高的书
SELECT title, price 
FROM books 
WHERE price > (SELECT AVG(price) FROM books);

-- 4.3 IN子查询 - 查询有借阅记录的书籍
SELECT title, stock
FROM books
WHERE book_id IN (
    SELECT DISTINCT book_id 
    FROM borrow_records 
    WHERE status IN ('borrowed', 'overdue')
);

-- 4.4 EXISTS子查询 - 查询有借阅记录的读者
SELECT name, email
FROM readers r
WHERE EXISTS (
    SELECT 1 
    FROM borrow_records br 
    WHERE br.reader_id = r.reader_id
);

-- 4.5 使用CASE WHEN进行条件判断
SELECT 
    title,
    price,
    CASE 
        WHEN price < 30 THEN '便宜'
        WHEN price >= 30 AND price < 40 THEN '适中'
        ELSE '昂贵'
    END AS price_level
FROM books;
```

### 7️⃣ 数据更新与删除 (UPDATE & DELETE)

```sql
-- ============================================
-- 练习5: 更新和删除数据
-- ============================================

-- 5.1 更新价格 - 所有书籍涨价10%
UPDATE books 
SET price = price * 1.1 
WHERE category = '科幻';

-- 5.2 更新库存 - 借出后减少库存
UPDATE books 
SET stock = stock - 1 
WHERE book_id = 1;

-- 5.3 更新借阅记录 - 归还书籍
UPDATE borrow_records 
SET return_date = CURRENT_DATE, 
    status = 'returned' 
WHERE record_id = 1;

-- 5.4 删除数据 - 删除过期读者(示例)
DELETE FROM readers 
WHERE status = 'expired';

-- 注意：由于外键约束，删除作者会级联删除其书籍
DELETE FROM authors WHERE author_id = 5;
```

### 8️⃣ 事务操作 (TRANSACTION)

```sql
-- ============================================
-- 练习6: 事务处理
-- ============================================

-- 借书事务：必须同时更新借阅表和库存
BEGIN TRANSACTION;

-- 插入借阅记录
INSERT INTO borrow_records (book_id, reader_id, due_date) 
VALUES (1, 3, DATE('now', '+14 days'));

-- 更新库存
UPDATE books 
SET stock = stock - 1 
WHERE book_id = 1 AND stock > 0;

-- 检查是否成功
SELECT changes();

-- 提交事务
COMMIT;

-- 或回滚事务
-- ROLLBACK;
```

### 9️⃣ 视图 (VIEW) 和索引

```sql
-- ============================================
-- 练习7: 创建视图和索引
-- ============================================

-- 7.1 创建视图 - 当前借阅信息
CREATE VIEW current_borrows AS
SELECT 
    r.name AS reader_name,
    b.title AS book_title,
    a.name AS author,
    br.borrow_date,
    br.due_date,
    CASE 
        WHEN br.due_date < CURRENT_DATE THEN '已过期'
        ELSE '正常'
    END AS overdue_status
FROM borrow_records br
JOIN readers r ON br.reader_id = r.reader_id
JOIN books b ON br.book_id = b.book_id
JOIN authors a ON b.author_id = a.author_id
WHERE br.status IN ('borrowed', 'overdue');

-- 使用视图
SELECT * FROM current_borrows;

-- 7.2 查看索引
SELECT * FROM sqlite_master WHERE type = 'index';

-- 7.3 删除索引
DROP INDEX IF EXISTS idx_borrow_records_status;
```
